cmake_minimum_required(VERSION 3.14) # FetchContent needs 3.11+
project(vscodium_cpp_tui)

set(CMAKE_CXX_STANDARD 20)

# --- This part replaces 'find_package' ---
include(FetchContent)

FetchContent_Declare(
  ftxui
  GIT_REPOSITORY https://github.com/ArthurSonzogni/ftxui
  GIT_TAG v5.0.0 # Use the latest stable version
)

FetchContent_MakeAvailable(ftxui)
# -----------------------------------------

add_executable(vscodium_cpp_tui main.cpp)

# Link the libraries (Note: the names are the same)
target_link_libraries(vscodium_cpp_tui PRIVATE
    ftxui::screen
    ftxui::dom
    ftxui::component
)
